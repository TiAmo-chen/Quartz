class Solution {

public:

    bool isValid(string s) 

    {

        if(s.length()%2!=0)

            return false;

  

        stack <char> a;

  

        for(int i = 0;i<s.length();i++)

        {

            //入栈

            if(s[i]=='{' || s[i]=='[' || s[i] =='(') a.push(s[i]);

  

            if(a.empty())

                return false;

            if(s[i]=='}')

            {

  

                if(a.top()!='{') return false;

                if(a.top()=='{') a.pop();

            }

  

            if(s[i]==']')

            {

                if(a.top()!='[') return false;

                if(a.top()=='[') a.pop();

            }

  

            if(s[i]==')')

            {

                if(a.top()!='(') return false;

                if(a.top()=='(') a.pop();

            }

  

        }

        return a.empty();

    }

};

class Solution {

public:

    bool isValid(string s) {

        stack<char> sc;

        for(char a:s){

            if(sc.size() && a==')' && sc.top()=='(')

                sc.pop();

            else if(sc.size() && a==']' && sc.top()=='[')

                sc.pop();

            else if(sc.size() && a=='}' && sc.top()=='{')

                sc.pop();

            else 

                sc.push(a);

        }

        return sc.empty();

    }

};
